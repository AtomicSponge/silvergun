30 // Split arguments into a vector of strings.
31 void split_args(const std::string& a) {
32 if(a == "") args.push_back("");
33 else {
34 std::stringstream arg_stream(a);
35 std::string segment;
36
37 while(std::getline(arg_stream, segment, ';')) {
38 args.push_back(segment);
39 }
40 }
