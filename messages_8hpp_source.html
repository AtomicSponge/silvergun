53 /*
54 * Process dispatcher components.
55 * Get messages for the entities and pass to each.
56 * Keeps checking for responces and will process as well.
57 */
58 static void dispatch(void) {
59 component_container<cmp::dispatcher> dispatch_components =
60 mgr::world::set_components<cmp::dispatcher>();
61
62 while (true) { // Infinite loop to verify all current messages are processed.
63 message_container temp_msgs = get("entities");
64 if (temp_msgs.empty()) break; // No messages, end while(true) loop.
65
66 // For all messages, check each dispatch component.
67 for (auto& m_it: temp_msgs) {
68 for (auto& c_it: dispatch_components) {
69 try {
70 if (m_it.get_to() == mgr::world::get_name(c_it.first)) {
71 c_it.second->handle_msg(c_it.first, m_it);
72 break; // Found, stop checking dispatch components.
73 }
74 } catch(const std::exception& e) {
75 throw e;
76 break;
77 } catch (...) { break; }
78 }
79 } // End double for
80 }
81 };

83 /*
84 * Load a new data file into the message queue.
85 * This is called when a new game is created.
86 * Events are placed in order according to the timer value.
87 */
88 static void load_file(const std::string& fname) {
89 _messages.clear();
90 // Open data file - read binary mode.
91 ALLEGRO_FILE* file;
92 file = al_fopen(fname.c_str(), "rb");
93 // File not found, error.
94 if (!file) {
95 al_fclose(file);
96 throw engine_error("Can not load game data!");
97 }
98
99 // Loop through the entire data file loading into the queue.
100 while (true) {
101 if (al_feof(file)) break; // End loop if eof.
102
103 int64_t timer = -1;
104 std::string sys, to, from, cmd, args;
105
106 try{
107 // Read the message from file.
108 read(*file, timer, sys, to, from, cmd, args);
109 // Add message to queue. Ignore incomplete messages.
110 if (sys != "" && cmd != "") _messages.push_back(message(timer, sys, to, from, cmd, args));
111 } catch (const std::exception& e) {
112 throw engine_error("Unable to read game data file!");
113 }
114 }
115 al_fclose(file);
116
117 // Sort the queue so timed events are in order first to last.
118 std::sort(_messages.begin(), _messages.end());
119 };

121 /*
122 * Get messages based on their command.
123 * Once events in the future are reached, break early.
124 */
125 static const message_container get(const std::string& sys) {
126 message_container temp_messages;
127 for (auto it = _messages.begin(); it != _messages.end();) {
128 // End early if events are in the future
129 if (it->get_timer() > engine_time::check()) break;
130 if ((it->get_timer() == -1 || it->get_timer() == engine_time::check()) && it->get_sys() == sys) {
131 if constexpr (build_options.debug_mode) log(*it, false);
132 temp_messages.push_back(*it); // Add the message to the temp vector to be returned.
133 it = _messages.erase(it); // Erase the message once processed.
134 } else it++; // Message not processed, iterate to next.
135 }
136 return temp_messages;
137 };

139 // Deletes timed messages that were not processed.
140 static void prune(void) {
141 for (auto it = _messages.begin(); it != _messages.end();) {
142 // End early if events are in the future.
143 if (it->get_timer() > engine_time::check()) break;
144 if constexpr (build_options.debug_mode) log(*it, true);
145 it = _messages.erase(it);
146 }
147 };

149 // Read a message from file.
150 static void read(
151 ALLEGRO_FILE& file,
152 int64_t& timer,
153 std::string& sys,
154 std::string& to,
155 std::string& from,
156 std::string& cmd,
157 std::string& args
158 ) {
159 // Read in timer value.
160 al_fread(&file, &timer, sizeof(int64_t));
161
162 // For each of the strings above, loop through and read
163 // each charecter, ending if we reach null termed or eof.
164 while (true) {
165 const char ch = al_fgetc(&file);
166 if (ch == '\0') break; // End loop if null terminated.
167 if (ch == EOF) {
168 throw engine_exception("Bad message file format!", "Messages", 2);
169 return;
170 }
171 sys += ch;
172 }
173 while (true) {
174 const char ch = al_fgetc(&file);
175 if (ch == '\0') break; // End loop if null terminated.
176 if (ch == EOF) {
177 throw engine_exception("Bad message file format!", "Messages", 2);
178 return;
179 }
180 to += ch;
181 }
182 while (true) {
183 const char ch = al_fgetc(&file);
184 if (ch == '\0') break; // End loop if null terminated.
185 if (ch == EOF) {
186 throw engine_exception("Bad message file format!", "Messages", 2);
187 return;
188 }
189 from += ch;
190 }
191 while (true) {
192 const char ch = al_fgetc(&file);
193 if (ch == '\0') break; // End loop if null terminated.
194 if (ch == EOF) {
195 throw engine_exception("Bad message file format!", "Messages", 2);

206 }
207 args += ch;

223 // Write a message to the debug log file
224 static void log(const message& msg, const bool& deleted) {
225 (deleted ?
226 debug_log_file << "MESSAGE DELETED | " :
227 debug_log_file << "PROC AT: " << engine_time::check() << " | ");
228 debug_log_file << "TIMER: " << msg.get_timer() << " | ";
229 debug_log_file << "SYS: " << msg.get_sys() << " | ";
230 if ((msg.get_to() != "") || (msg.get_from() != "")) {
231 debug_log_file << "TO: " << msg.get_to() << " | ";
232 debug_log_file << "FROM: " << msg.get_from() << " | ";
233 }
234 debug_log_file << "CMD: " << msg.get_cmd() << " | ";
235 debug_log_file << "ARGS: ";
236 msg_args arglist = msg.get_args();
237 for (auto i = arglist.begin(); i != arglist.end(); i++) {
238 debug_log_file << *i;
239 if (std::next(i, 1) != arglist.end()) debug_log_file << ";";
240 }
241 debug_log_file << std::endl;
242 };

269 static bool load_script(const std::string& fname) {
270 // Open data file - read binary mode.
271 ALLEGRO_FILE* file;
272 file = al_fopen(fname.c_str(), "rb");
273 // File not found, error.
274 if (!file) {
275 al_fclose(file);
276 return false;

279 // Loop through the entire data file loading into the queue.

281 if (al_feof(file)) break; // End loop if eof.
282

284 std::string sys, to, from, cmd, args;
285
286 try {
287 // Read the message from file.
288 read(*file, timer, sys, to, from, cmd, args);
289 // Add the current time to the timer value.
290 if (timer != -1) timer += engine_time::check();
291 // Add message to queue. Ignore incomplete messages. Sort while adding.
292 if (sys != "" && cmd != "") add(message(timer, sys, to, from, cmd, args));
293 } catch(const std::exception& e) { throw e; }
294 }
295 al_fclose(file);
