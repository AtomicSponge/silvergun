118 static entity_id new_entity(void) {
119 entity_id next_id;
120
121 if (entity_counter == ENTITY_MAX) { // Counter hit max.
122 bool test = false;
123 // Look for the first available ID.
124 for (next_id = ENTITY_START; !test; next_id++) {
125 if (next_id == ENTITY_MAX) return ENTITY_ERROR; // No available ID, error.
126 // See if the new ID does not exist.
127 test = (std::find_if (entity_vec.begin(), entity_vec.end(), [&next_id](const entity& e){ return e.first == next_id; }) == entity_vec.end());
128 }
129 } else { // Counter not max, use the counter for entity ID.
130 next_id = entity_counter;
131 entity_counter++;
132 }
133
134 // Set a new name. Make sure name doesn't exist.
135 std::string entity_name = "Entity" + std::to_string(next_id);
136 bool test = false;
137 for (entity_id temp_id = ENTITY_START; !test; temp_id++) {
138 if (temp_id == ENTITY_MAX) return ENTITY_ERROR; // Couldn't name entity, error.
139 // See if the new name does not exist.
140 test = (std::find_if (entity_vec.begin(), entity_vec.end(), [&entity_name](const entity& e){ return e.second == entity_name; }) == entity_vec.end());
141 // If it does, append the temp number and try that.
142 if (!test) entity_name = "Entity" + std::to_string(next_id) + std::to_string(temp_id);
143 }
144
145 // Tests complete, insert new entity.
146 entity_vec.push_back(std::make_pair(next_id, entity_name));
147 return next_id; // Return new entity ID.
148 };

279 template <typename T, typename... Args>
280 inline static bool add_component(
281 const entity_id& e_id,
282 Args... args
283 ) {
284 if (!entity_exists(e_id)) return false;
285
286 // Check derived types of existing components, make sure one does not already exist.
287 const auto check_entity = get_entity(e_id);
288
289 for (auto& it: check_entity) {
290 auto& r = *it.get();
291 if (typeid(r).name() == typeid(T).name()) return false;
292 }
293
294 _world.insert(std::make_pair(e_id, std::make_shared<T>(args...)));
295 return true;
296 };
