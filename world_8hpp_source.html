118 static entity_id new_entity(void) {
119 entity_id next_id;
120
121 if (entity_counter == ENTITY_MAX) { // Counter hit max.
122 bool test = false;
123 // Look for the first available ID.
124 for (next_id = ENTITY_START; !test; next_id++) {
125 if (next_id == ENTITY_MAX) return ENTITY_ERROR; // No available ID, error.
126 // See if the new ID does not exist.
127 test = (std::find_if (entity_vec.begin(), entity_vec.end(),
128 [&next_id](const entity& e){ return e.first == next_id; })
129 == entity_vec.end());
130 }
131 } else { // Counter not max, use the counter for entity ID.
132 next_id = entity_counter;
133 entity_counter++;
134 }
135
136 // Set a new name. Make sure name doesn't exist.
137 std::string entity_name = "Entity" + std::to_string(next_id);
138 bool test = false;
139 for (entity_id temp_id = ENTITY_START; !test; temp_id++) {
140 if (temp_id == ENTITY_MAX) return ENTITY_ERROR; // Couldn't name entity, error.
141 // See if the new name does not exist.
142 test = (std::find_if (entity_vec.begin(), entity_vec.end(),
143 [&entity_name](const entity& e){ return e.second == entity_name; })
144 == entity_vec.end());
145 // If it does, append the temp number and try that.
146 if (!test) entity_name = "Entity" + std::to_string(next_id) + std::to_string(temp_id);
147 }
148
149 // Tests complete, insert new entity.
150 entity_vec.push_back(std::make_pair(next_id, entity_name));
151 return next_id; // Return new entity ID.
152 };

293 template <typename T, typename... Args>
294 inline static bool add_component(
295 const entity_id& e_id,
296 Args... args
297 ) {
298 if (!entity_exists(e_id)) return false;
299
300 // Check derived types of existing components, make sure one does not already exist.
301 const auto check_entity = get_entity(e_id);
302
303 for (auto& it: check_entity) {
304 auto& r = *it.get();
305 if (typeid(r).name() == typeid(T).name()) return false;
306 }
307
308 _world.insert(std::make_pair(e_id, std::make_shared<T>(args...)));
309 return true;
310 };
