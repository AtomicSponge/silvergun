38 // Takes spawner messages and processes.
39 static void process_messages(const message_container& messages) {
40 for (auto& m_it: messages) {
41 if (m_it.get_cmd() == "new") {
42 auto s_it = spawns.find(m_it.get_arg(0));
43 if (s_it != spawns.end())
44 // Make sure the number of arguments match what's expected.
45 // Note that we do not count the first argument.
46 if (m_it.num_args() == s_it->second.first + 1) {
47 try {
48 s_it->second.second(mgr::world::new_entity(), m_it.get_args());
49 } catch(const std::exception& e) { throw e; }
50 }
51 }
52
53 if (m_it.get_cmd() == "delete") {
54 entity_id delete_entity_id = mgr::world::get_id(m_it.get_arg(0));
55 if (delete_entity_id != slv::mgr::ENTITY_ERROR) {
56 mgr::world::delete_entity(delete_entity_id);
57 }
58 }
59 }
60 };

93 static bool remove(const std::string& name) {
94 auto it = std::find_if (spawns.begin(), spawns.end(),
95 [&name](const auto& e){ return e.first == name; });
96 if (it != spawns.end()) {
