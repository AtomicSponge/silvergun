97 /*
98 * Main engine loop (single pass)
99 */
100 static void main_loop(void) {
101 input::check_events(); // Check for input.
102
103 // Game not running, make sure the timer isn't.
104 if(!config::flags::engine_started) al_stop_timer(main_timer);
105 else {
106 // Pause / resume timer check. Also process the on_pause events.
107 if(config::flags::engine_paused && al_get_timer_started(main_timer)) {
108 al_stop_timer(main_timer);
109 on_engine_pause();
110 }
111 if(!config::flags::engine_paused && !al_get_timer_started(main_timer)) {
112 on_engine_unpause();
113 al_resume_timer(main_timer);
114 }
115 }
116
117 ALLEGRO_EVENT event;
118 if(al_get_next_event(main_event_queue, &event)) {
119 switch(event.type) {
120 // Call our game logic update on timer events.
121 // Timer is only running when the game is running.
122 case ALLEGRO_EVENT_TIMER:
123 // Set the engine_time object to the current time.
124 engine_time::set(al_get_timer_count(main_timer));
125 // Run all systems.
126 mgr::systems::run();
127 // Process messages.
128 mgr::messages::dispatch();
129 // Get any spawner messages and pass to handler.
130 mgr::spawner::process_messages(mgr::messages::get("spawner"));
131 break;
132 // Check if display looses focus.
133 case ALLEGRO_EVENT_DISPLAY_SWITCH_OUT:
134 out_of_focus();
135 break;
136 // Check if display returns to focus.
137 case ALLEGRO_EVENT_DISPLAY_SWITCH_IN:
138 back_in_focus();
139 break;
140 // Force quit if the game window is closed.
141 case ALLEGRO_EVENT_DISPLAY_CLOSE:
142 if(config::flags::engine_started) process_end_game(true);
143 config::_flags::is_running = false;
144 break;
145 // Window has been resized.
146 case ALLEGRO_EVENT_DISPLAY_RESIZE:
147 al_acknowledge_resize(_display);
148 break;
149 }
150 }
151
152 // Render the screen.
153 mgr::gfx::renderer::render();
154 // Get any system messages and pass to handler.
155 cmds.process_messages(mgr::messages::get("system"));
156 // Send audio messages to the audio queue.
157 mgr::audio::process_messages(mgr::messages::get("audio"));
158 // Delete unprocessed messages.
159 mgr::messages::prune();
160 };

162 /*
163 * Call to start a new game.
164 * Loads a game data file and user defined systems and starting entities.
165 * Gets passed game data file to load.
166 */
167 static void process_new_game(const std::string& game_script) {
168 std::cout << "Starting new game... ";
169 std::srand(std::time(nullptr)); // Seed random, using time.
170
171 // Load a new message data file.
172 if(!game_script.empty()) mgr::messages::load_file(game_script);
173
174 // Load systems and prevent further systems from being loaded.
175 load_systems();
176 mgr::systems::finalized = true;
177 if(mgr::systems::empty()) throw engine_error("No systems have been loaded!");
178
179 // Stop audio manager from playing sounds.
180 mgr::audio::music::a::stop();
181 mgr::audio::music::b::stop();
182 mgr::audio::ambiance::stop();
183 mgr::audio::voice::stop();
184 mgr::audio::sample::clear_instances();
185
186 // Clear world and load starting entities.
187 mgr::world::clear();
188
189 // Call custom start game process
190 new_game();
191
192 // Restart the timer at zero.
193 al_stop_timer(main_timer);
194 al_set_timer_count(main_timer, 0);
195 engine_time::set(al_get_timer_count(main_timer));
196 config::_flags::engine_started = true;
197 config::flags::engine_paused = false;
198 al_start_timer(main_timer);
199 std::cout << "READY!\n";
200 };
201
202 /*
203 * Call to end the game.
204 * Clears out the entities and systems and runs user defined end process.
205 * If passed true, skips the custom game cleanup.
206 */
207 static void process_end_game(const bool& force) {
208 std::cout << "Ending game... ";
209 al_stop_timer(main_timer);
210 config::_flags::engine_started = false;
211 al_set_timer_count(main_timer, 0);
212 engine_time::set(al_get_timer_count(main_timer));
213
214 // Stop audio manager from playing sounds.
215 mgr::audio::music::a::stop();
216 mgr::audio::music::b::stop();
217 mgr::audio::ambiance::stop();
218 mgr::audio::voice::stop();
219 mgr::audio::sample::clear_instances();
220
221 // Call custom end game process.
222 if(!force) end_game();
223
224 // Clear managers.
225 mgr::world::clear();
226 mgr::systems::clear();
227 mgr::messages::clear();
228 std::cout << "DONE!\n";
229 };

265 if(initialized == true) throw engine_error(display::window_title + " already running!");
266 initialized = true;

277 std::cout << "OK!\n";
278 config::_flags::audio_installed = al_install_audio();
279 // Input detection.

294 al_set_physfs_file_interface();
295 std::cout << "OK!\n";

402 static void de_init(void) {
403 std::cout << "\nStopping WTEngine...\n";
404 PHYSFS_deinit();
405
406 std::cout << "Cleaning up engine objects... ";
407 al_destroy_timer(main_timer);
408 al_destroy_event_queue(main_event_queue);
409 input::destroy_event_queue();
410 destroy_display();
411 al_inhibit_screensaver(false);
412 std::cout << "OK!\n";

419
420 if constexpr (build_options.debug_mode) {
421 logger::stop();
422 mgr::messages::message_log_stop();
423 }
424
425 initialized = false;
426 std::cout << "\nGood bye!\n\n";
427 };
