67 /*
68 * Main engine loop (single pass)
69 */
70 static void main_loop(void) {
71 input::check_events(); // Check for input.
72
73 // Pause / resume timer check. Also process the on_pause events.
74 if (config::flags::engine_paused && al_get_timer_started(main_timer)) {
75 al_stop_timer(main_timer);
76 on_engine_pause();
77 }
78 if (!config::flags::engine_paused && !al_get_timer_started(main_timer)) {
79 on_engine_unpause();
80 al_resume_timer(main_timer);
81 }
82
83 ALLEGRO_EVENT event;
84 if (al_get_next_event(main_event_queue, &event)) {
85 switch(event.type) {
86 // Call our game logic update on timer events.
87 // Timer is only running when the game is running.
88 case ALLEGRO_EVENT_TIMER:
89 // Set the engine_time object to the current time.
90 engine_time::set(al_get_timer_count(main_timer));
91 // Run all systems.
92 mgr::systems::run();
93 // Process messages.
94 mgr::messages::dispatch();
95 // Get any spawner messages and pass to handler.
96 mgr::spawner::process_messages(mgr::messages::get("spawner"));
97 break;
98 // Check if display looses focus.
99 case ALLEGRO_EVENT_DISPLAY_SWITCH_OUT:
100 out_of_focus();
101 break;
102 // Check if display returns to focus.
103 case ALLEGRO_EVENT_DISPLAY_SWITCH_IN:
104 back_in_focus();
105 break;
106 // Force quit if the game window is closed.
107 case ALLEGRO_EVENT_DISPLAY_CLOSE:
108 deinitialize();
109 break;
110 // Window has been resized.
111 case ALLEGRO_EVENT_DISPLAY_RESIZE:
112 al_acknowledge_resize(_display);
113 break;
114 }
115 }
116
117 if (current_scene != nullptr) current_scene->loop();
118
119 // Render the screen.
120 mgr::gfx::renderer::render();
121 // Send audio messages to the audio queue.
122 mgr::audio::process_messages(mgr::messages::get("audio"));
123 // Delete unprocessed messages.
124 mgr::messages::prune();
125 };

147 static void initialize(int width, int height) {
148 std::cout << "Starting WTEngine...\n";
149 if (initialized == true) throw engine_error(display::window_title + " already running!");
150 initialized = true;
151
152 std::cout << "Loading Allegro Game Library... ";
153 // Initialize Allegro.
154 if (!al_init()) throw engine_error("Allegro failed to load!");
155 std::cout << "OK!\n";
156
157 // Initialize additional Allegro components.
158 std::cout << "Loading Allegro add-ons... ";
159 if (!al_init_image_addon()) throw engine_error("Failed to load Allegro image addon!");
160 if (!al_init_font_addon()) throw engine_error("Failed to load Allegro font addon!");
161 std::cout << "OK!\n";
162 config::_flags::audio_installed = al_install_audio();
163 // Input detection.
164 if constexpr (build_options.keyboard_enabled)
165 config::_flags::keyboard_installed = al_install_keyboard();
166 if constexpr (build_options.mouse_enabled)
167 config::_flags::mouse_installed = al_install_mouse();
168 if constexpr (build_options.joystick_enabled)
169 config::_flags::joystick_installed = al_install_joystick();
170 if constexpr (build_options.touch_enabled)
171 config::_flags::touch_installed = al_install_touch_input();
172
173 // Configure display. Called from wte_display class.
174 std::cout << "Configuring display... ";
175 create_display(width, height);
176 std::cout << "OK!\n";
177
178 // Disable pesky screensavers.
179 al_inhibit_screensaver(true);
180
181 std::cout << "Creating main timer and event queue... ";
182 // Configure main timer.
183 main_timer = al_create_timer(1.0f / build_options.ticks_per_sec);
184 if (!main_timer) throw engine_error("Failed to create timer!");
185 // Configure main event queue.
186 main_event_queue = al_create_event_queue();
187 if (!main_event_queue) throw engine_error("Failed to create main event queue!");
188 std::cout << "OK!\n";
189
190 // Register event sources.
191 al_register_event_source(main_event_queue, al_get_display_event_source(_display));
192 al_register_event_source(main_event_queue, al_get_timer_event_source(main_timer));
193
194 // Create the input event queue
195 input::create_event_queue();
196
197 // Allegro extras
198 al_init_primitives_addon();
199 if (config::flags::audio_installed) al_init_acodec_addon();
200
201 // Load systems and prevent further systems from being loaded.
202 std::cout << "Loading systems... ";
203 load_systems();
204 mgr::systems::finalized = true;
205 if (mgr::systems::empty()) throw engine_error("No systems have been loaded!");
206 std::cout << "OK!\n";
207
208 std::cout << "Loading audio... ";
209 mgr::audio::initialize();
210 std::cout << "OK!\n";
211
212 std::cout << "Initializing graphics... ";
213 mgr::gfx::renderer::set_viewport_size(width, height);
214 mgr::gfx::renderer::initialize();
215 std::cout << "OK!\n";
216
217 // Generate Allegro's default font and load into asset mgr.
218 mgr::assets::load<ALLEGRO_FONT>("wte_default_font", make_asset<ALLEGRO_FONT>());
219
220 if constexpr (build_options.debug_mode) {
221 mgr::messages::message_log_start();
222 logger::start();
223 }
224
225 config::flags::engine_paused = false;
226 config::_flags::is_running = true;
227 std::cout << "Engine started successfully!\n\n";
228
229 al_stop_timer(main_timer);
230 al_set_timer_count(main_timer, 0);
231 engine_time::set(al_get_timer_count(main_timer));
232 al_start_timer(main_timer);
233 };

238 static void deinitialize(void) {
239 std::cout << "\nStopping WTEngine...\n";
240
241 config::_flags::is_running = false;
242
243 mgr::audio::de_init();
244 mgr::gfx::renderer::de_init();
245 mgr::assets::clear_al_objects();
246
247 std::cout << "Cleaning up engine objects... ";
248 al_destroy_timer(main_timer);
249 al_destroy_event_queue(main_event_queue);
250 input::destroy_event_queue();
251 destroy_display();
252 al_inhibit_screensaver(false);
253 std::cout << "OK!\n";
254 std::cout << "Stopping Allegro... ";
255 al_shutdown_font_addon();
256 if (config::flags::audio_installed) al_uninstall_audio();
257 al_shutdown_primitives_addon();
258 al_uninstall_system();
259 std::cout << "OK!\n";
260
261 if constexpr (build_options.debug_mode) {
262 logger::stop();
263 mgr::messages::message_log_stop();
264 }
265
266 initialized = false;
267 std::cout << "\nGood bye!\n\n";
268 };

281 static void load_scene(const std::string& name) {
282 mgr::world::clear();
283
284 if (current_scene != nullptr) current_scene->unload();
