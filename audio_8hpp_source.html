82 // Map the audio commands.
83 // Mixer 1
84 cmds.add("music-loop", 2, [](const msg_args& args) {
85 if (args[0] == "a") {
86 if (args[1] == "disable") audio::music::a::loop(false);
87 else audio::music::a::loop(true);
88 }
89 if (args[0] == "b") {
90 if (args[1] == "disable") audio::music::b::loop(false);
91 else audio::music::b::loop(true);
92 }
93 });
94 cmds.add("music-play", 2, [](const msg_args& args) {
95 if (args[0] == "a")
96 audio::music::a::play(mgr::assets::get<ALLEGRO_AUDIO_STREAM>(args[1]));
97 if (args[0] == "b")
98 audio::music::b::play(mgr::assets::get<ALLEGRO_AUDIO_STREAM>(args[1]));
99 });
100 cmds.add("music-stop", 1, [](const msg_args& args) {
101 if (args[0] == "a") audio::music::a::stop();
102 if (args[0] == "b") audio::music::b::stop();
103 });
104 cmds.add("music-pause", 1, [](const msg_args& args) {
105 if (args[0] == "a") audio::music::a::pause();
106 if (args[0] == "b") audio::music::b::pause();
107 });
108 cmds.add("music-unpause", 1, [](const msg_args& args) {
109 if (args[0] == "a") audio::music::a::unpause();
110 if (args[0] == "b") audio::music::a::unpause();
111 });
112 // Mixer 2
113 cmds.add("sample-play", 2, [](const msg_args& args) {
114 float gain = 1.0f;
115 float pan = ALLEGRO_AUDIO_PAN_NONE;
116 float speed = 1.0f;
117
118 if (args.size() >= 3) {
119 gain = std::stof(args[2]);
120 if (gain < 0.0f || gain > 1.0f) gain = 1.0f;
121 }
122 if (args.size() >= 4) {
123 pan = std::stof(args[3]);
124 if (pan < -1.0f || pan > 1.0f) pan = ALLEGRO_AUDIO_PAN_NONE;
125 }
126 if (args.size() >= 5) {
127 speed = std::stof(args[4]);
128 if (speed <= 0.0f || speed > 2.0f) speed = 1.0f;
129 }
130 audio::sample::play(
131 mgr::assets::get<ALLEGRO_SAMPLE>(args[0]),
132 args[1], gain, pan, speed
133 );
134 });
135 cmds.add("sample-stop", 1, [](const msg_args& args) {
136 audio::sample::stop(args[0]);
137 });
138 cmds.add("sample-clear-instances", 0, [](const msg_args& args) {
139 audio::sample::clear_instances();
140 });
141 // Mixer 3
142 cmds.add("voice-play", 1, [](const msg_args& args) {
143 audio::voice::play(mgr::assets::get<ALLEGRO_AUDIO_STREAM>(args[0]));
144 });
145 cmds.add("voice-stop", 0, [](const msg_args& args) {
146 audio::voice::stop();
147 });
148 cmds.add("voice-pause", 0, [](const msg_args& args) {
149 audio::voice::pause();
150 });
151 cmds.add("voice-unpause", 0, [](const msg_args& args) {

451 static void play(
452 wte_asset<ALLEGRO_SAMPLE> sample,

464 static void play(
465 wte_asset<ALLEGRO_SAMPLE> sample,
466 const std::string& ref,
467 const float& gain,
468 const float& pan,
469 const float& speed
470 ) {
471 if (ref == "once") {
472 // Play the sample once.
473 al_play_sample(sample.get(), gain, pan, speed, ALLEGRO_PLAYMODE_ONCE, NULL);
474 } else {
475 // If the reference is already playing, end.
476 if (sample_instances.find(ref) != sample_instances.end()) return;
477 // Store playing reference
478 ALLEGRO_SAMPLE_ID temp_sample_id;
479 if (al_play_sample(sample.get(), gain, pan, speed, ALLEGRO_PLAYMODE_LOOP, &temp_sample_id))
480 sample_instances.insert(std::make_pair(ref, temp_sample_id));
481 }
482 };
483
488 static void stop(const std::string& ref) {
489 // If instance does not exist, end.
490 if (sample_instances.find(ref) == sample_instances.end()) return;
491 al_stop_sample(&sample_instances.find(ref)->second);
492 sample_instances.erase(sample_instances.find(ref));
493 };
494
498 static void clear_instances(void) {
499 for (auto sample_instance = sample_instances.begin(); sample_instance != sample_instances.end();) {
500 al_stop_sample(&sample_instance->second);
501 sample_instances.erase(sample_instance);
502 sample_instance = sample_instances.begin();
503 }
504 };

513 config::_volume::sample = 0.0f);
514 set_volume();
