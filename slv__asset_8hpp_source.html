89template <typename T>
90inline static const slv_asset<T> make_asset(const std::string& fname) {
91 static_assert(std::is_same_v<T, ALLEGRO_BITMAP> ||
92 std::is_same_v<T, ALLEGRO_SAMPLE> ||
93 std::is_same_v<T, ALLEGRO_AUDIO_STREAM>, "Must be an Allegro type to load by filename.");
94
95 if constexpr (std::is_same_v<T, ALLEGRO_BITMAP>) {
96 std::shared_ptr<ALLEGRO_BITMAP> temp_ptr(al_load_bitmap(fname.c_str()), al_destroy_bitmap);
97 return temp_ptr;
98 }
99
100 if constexpr (std::is_same_v<T, ALLEGRO_SAMPLE>) {
101 std::shared_ptr<T> temp_ptr(al_load_sample(fname.c_str()), al_destroy_sample);
102 return temp_ptr;
103 }
104
105 if constexpr (std::is_same_v<T, ALLEGRO_AUDIO_STREAM>) {
106 std::shared_ptr<T> temp_ptr(al_load_audio_stream(fname.c_str(), 4, 2048), al_destroy_audio_stream);
107 return temp_ptr;
108 }
109};
