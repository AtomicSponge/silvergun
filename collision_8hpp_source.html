27 void run(void) override {
28 const_component_container<cmp::hitbox> hitbox_components =
29 mgr::world::get_components<cmp::hitbox>();
30
31 for (auto& it_a: hitbox_components) {
32 for (auto& it_b: hitbox_components) {
33 cmp::const_comp_ptr<cmp::location> temp_location_a = mgr::world::get_component<cmp::location>(it_a.first);
34 cmp::const_comp_ptr<cmp::location> temp_location_b = mgr::world::get_component<cmp::location>(it_b.first);
35
36 /*
37 * Only test if: Not the same entity.
38 * Entities are on different teams.
39 * Both entities are solid.
40 */
41 if (
42 it_a.first != it_b.first &&
43 it_a.second->team != it_b.second->team &&
44 it_a.second->solid && it_b.second->solid
45 ) {
46 // Use AABB to test colision
47 if (
48 temp_location_a->pos_x < temp_location_b->pos_x + it_b.second->width &&
49 temp_location_a->pos_x + it_a.second->width > temp_location_b->pos_x &&
50 temp_location_a->pos_y < temp_location_b->pos_y + it_b.second->height &&
51 temp_location_a->pos_y + it_a.second->height > temp_location_b->pos_y
52 ) {
53 // Send a message that two entities colided.
54 // Each entity will get a colision message.
55 // Ex: A hit B, B hit A.
56 mgr::messages::add(message("entities", mgr::world::get_name(it_a.first), mgr::world::get_name(it_b.first), "colision", "")
57 );
58 }
59 } // End skip self check
60 } // End it_b loop
61 } // End it_a loop
62 };
