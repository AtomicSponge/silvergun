97 // Configure the display.
98 static void create_display(int w, int h) {
99 // Set a default screen size
100 config::_gfx::screen_w = w;
101 config::_gfx::screen_h = h;
102
103 al_reset_new_display_options();
104
105 // Configure vsync options. Gfx driver may override this.
106 // 0 - System decides | 1 - VSync on | 2 - VSync off
107 if (config::gfx::vsync >= 0 && config::gfx::vsync <= 2) {
108 al_set_new_display_option(ALLEGRO_VSYNC, config::gfx::vsync, ALLEGRO_SUGGEST);
109 } else {
110 al_set_new_display_option(ALLEGRO_VSYNC, 0, ALLEGRO_SUGGEST);
111 }
112
113 // Check if a display mode is set.
114 if (config::gfx::display_mode == 1) {
115 (build_options.opengl_latest ?
116 al_set_new_display_flags(ALLEGRO_OPENGL_3_0 | ALLEGRO_FULLSCREEN_WINDOW):
117 al_set_new_display_flags(ALLEGRO_OPENGL | ALLEGRO_FULLSCREEN_WINDOW));
118 } else {
119 (build_options.opengl_latest ?
120 al_set_new_display_flags(ALLEGRO_OPENGL_3_0 | ALLEGRO_WINDOWED):
121 al_set_new_display_flags(ALLEGRO_OPENGL | ALLEGRO_WINDOWED));
122 }
123
124 // Create the display. Full screen windowed defaults to the display resolution.
125 _display = al_create_display(config::gfx::screen_w, config::gfx::screen_h);
126
127 // Display failed to load, try a fallback.
128 if (!_display) {
129 (build_options.opengl_latest ?
130 al_set_new_display_flags(ALLEGRO_OPENGL_3_0 | ALLEGRO_WINDOWED):
131 al_set_new_display_flags(ALLEGRO_OPENGL | ALLEGRO_WINDOWED));
132 _display = al_create_display(config::gfx::screen_w,
133 config::gfx::screen_h);
134 if (!_display) throw engine_error("Failed to configure display!");
135 config::_gfx::display_mode = 0;
136 config::_gfx::scale_factor = 1.0f;
137 }
138
139 // Set window title.
140 al_set_window_title(_display, window_title.c_str());
141
142 // Set window icon.
143 ALLEGRO_FILE* file;
144 file = al_fopen("icon.bmp", "rb");
145 if (file) {
146 ALLEGRO_BITMAP* icon_bitmap = al_load_bitmap_f(file, ".bmp");
147 al_set_display_icon(_display, icon_bitmap);
148 al_destroy_bitmap(icon_bitmap);
149 }
150 al_fclose(file);
151 };
152
153 // Destroy the display.
154 static void destroy_display(void) {
155 al_destroy_display(_display);
156 };
